  .sel(time=ds.time.dt.month == 4) \
  .mean(dim="time")
tmp_april_2024 = ds["temperature_2m"] \
  .sel(time="2024-04") \
  .mean(dim="time")

(tmp_april_2024 - tmp_april_mean).plot()
